# Reality Share URI Example
# Copy this URI and import it directly into your client

# Replace placeholders before using:
# - YOUR_SERVER_IP: Your server's public IP address
# - REPLACE_WITH_YOUR_UUID: UUID generated with 'sing-box generate uuid'
# - REPLACE_WITH_YOUR_PUBLIC_KEY: Public key from 'sing-box generate reality-keypair'
# - REPLACE_WITH_YOUR_SHORT_ID: Short ID from 'openssl rand -hex 4'

vless://REPLACE_WITH_YOUR_UUID@YOUR_SERVER_IP:443?encryption=none&flow=xtls-rprx-vision&security=reality&sni=www.microsoft.com&fp=chrome&pbk=REPLACE_WITH_YOUR_PUBLIC_KEY&sid=REPLACE_WITH_YOUR_SHORT_ID&type=tcp&headerType=none#Reality-YOUR_SERVER_IP

# URI Format Breakdown:
# vless://          - Protocol
# UUID@IP:PORT      - User ID, server address and port
# encryption=none   - No additional encryption (Reality provides it)
# flow=xtls-rprx-vision - Vision protocol for better performance
# security=reality  - Reality TLS camouflage
# sni=www.microsoft.com - Server Name Indication
# fp=chrome         - TLS fingerprint (Chrome browser)
# pbk=PUBLIC_KEY    - Reality public key
# sid=SHORT_ID      - Reality short ID (8 hex chars)
# type=tcp          - Transport protocol
# #Reality-IP       - Connection name/remark
